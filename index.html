        <h1>Daily Hookah Log</h1>

        <p>Log the number of hookahs you smoke each day.</p>

        <button class="log-button" onclick="logHookah()">Log Hookah</button>

        <p id="count-display">You've smoked <span id="hookah-count">0</span> hookahs today.</p>

        <p id="date-display"></p>

    </div>


    <script>

        let hookahCount = 0;

        const today = new Date().toLocaleDateString();


        document.getElementById('date-display').innerText = `Today's date: ${today}`;


        function logHookah() {

            hookahCount += 1;

            document.getElementById('hookah-count').innerText = hookahCount;

            // Store count in local storage

            localStorage.setItem(today, hookahCount);

        }


        // Load previous count from local storage

        const savedCount = localStorage.getItem(today);

        if (savedCount) {

            hookahCount = parseInt(savedCount);

            document.getElementById('hookah-count').innerText = hookahCount;

        }

    </script>
